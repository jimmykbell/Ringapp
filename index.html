document.addEventListener("DOMContentLoaded", function() { // Retrieve all ring data from Firebase firebase.database().ref('rings').once('value', (snapshot) => { const ringsData = snapshot.val(); if (ringsData) { for (const ringId in ringsData) { const ringData = ringsData[ringId]; const ring = document.getElementById('ring' + ringId); ring.textContent = `Ring ${ringId}\n${ringData.status}\n${ringData.timestamp}`; ring.style.backgroundColor = ringData.color; } } }); });
